def introduce(name, class_name, school ="常州工学院"):  
    """生成介绍语，学校默认为常州工学院 """    introducing= f"大家好，我是{name}, 来自{school} {class_name},请大家多多关照！"  
    print(introducing)  
  
# 分析一下函数调用,是否正确  
introduce("张小斐", "23数媒2")    #调用1  
introduce("谭宇")    #调用2  
introduce("赵明","23数媒3")     #调用3  
introduce("潘明月","23数媒3","常州大学")     #调用4
